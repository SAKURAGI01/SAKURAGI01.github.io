    玩具代码

    通过实现简单功能， 初识游戏如何制作

    1. 初始化游戏，并建立游戏资源

    pygame.init()

    2. 创建显示窗口

    pygame.display.set_mode((w,h))

    3. 获取当前显示的display对象

    screen = pygame.display.get_surface()

    4. 载入背景图 并 缩放到高宽

    bgpic = pygame.image.load('图片名')
bgpic = pygame.transform.scale(bgpic,(w,h))

    5. 创建精灵对象

    mario = pygame.sprite.Sprite()

    6. 初始化精灵对象的 图片 和 矩阵坐标

    mario.image = mario_image
mario.rect = mario.image.get_rect()
mario.rect.x, mario.rect.y = w/2, h/2

    7. 定义精灵组 并 加入mario

    player_group = pygame.sprite.Group()
player_group.add(mario)

    8. 监视键盘和鼠标的事件

     for event in pygame.event.get():
    8.1 设置退出选项
    if event.type == pygame.QUIT:
pygame.display.quit()
quit()
    8.2 设置移动选项
    if event.type == pygame.KEYDOWN:
    keys = pygame.key.get_pressed()
        if keys[pygame.K_DOWN]:
            mario.rect.y += 10
        if keys[pygame.K_UP]:
            mario.rect.y -= 10
        if keys[pygame.K_RIGHT]:
            mario.rect.x += 10
        if keys[pygame.K_LEFT]:
            mario.rect.x -= 10

    9. 绘制图像至屏幕中

        screen.blit(bgpic,(0, 0))
    player_group.draw(screen)
    pygame.display.update()
