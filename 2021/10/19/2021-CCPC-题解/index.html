{
    int res = 1;

    for (; b; b >>= 1)
    {
        if (b & 1) res = (LL) res * a % p;
        a = (LL) a * a % p;
    }

    return res;
}

    #include <iostream>
#include <cstring>

using namespace std;

const int N = 1010;

char s[N], t[N];

int dp[N][N];


int main()
{
    scanf("%s%s", s + 1, t + 1);

    int n = strlen(s + 1);
    int m = strlen(t + 1);

    for (int j = 0; j <= m; j ++ ) dp[0][j] = 0;
    for (int i = 0; i <= n; i ++ ) dp[i][0] = 1;

    for (int i = 1; i <= n; i ++ )
    {
        for (int j = 1; j <= m; j ++ )
        {
            if (s[i] == t[j]) dp[i][j] = dp[i - 1][j] + dp[i - 1][j - 1];
            else dp[i][j] = dp[i - 1][j];
        }
    }

    cout << dp[n][m] << endl;

    return 0;
}
