    #include <iostream>
#include <cstring>
#include <queue>

using namespace std;

const int N = 1e5 + 10, M = 110;

int w[N], d[M][M];//d[][] è¡¨ç¤ºç¦» i æœ€è¿‘çš„ä¸”å±æ€§å€¼ä¸º j çš„ç‚¹çš„è·ç¦»

int e[N], h[N], ne[N], idx;

bool st[M];

int n, m, q;

void add(int a, int b)
{
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx ++ ;
}

void bfs(int s)
{
    queue<int> qq;

    memset(st, false, sizeof st);

    for (int i = 1; i <= n; i ++ )
        if (w[i] == s)
        {
            qq.push(i);
            st[i] = true;
            d[i][s] = 0;
        }

    while (qq.size())
    {
        int t = qq.front();
        qq.pop();

        for (int i = h[t]; i != -1; i = ne[i])
        {
            int j = e[i];
            if (st[j]) continue;
            d[j][s] = d[t][s] + 1;
            qq.push(j);
            st[j] = true;
        }
    }
}

int main()
{
    memset(h, -1, sizeof h);

    cin >> n >> m  >> q;

    for (int i = 1; i <= n; i ++ )
        scanf("%d", &w[i]);

    while (m -- )
    {
        int u, v;
        scanf("%d%d", &u, &v);

        if (u == v) continue;

        add(u, v), add(v, u);
    }

    memset(d, -1, sizeof d);

    for (int i = 1; i <= 100; i ++ ) bfs(i);

    while (q -- )
    {
        int p, a;
        scanf("%d%d", &p, &a);

        if (d[p][i] != -1) minn = min(minn, d[p][i]);
